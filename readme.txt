# Command
- git init // dự án có thể sử dụng được git sau khi đánh lệnh này, kiểu khởi tạo
- git status
- git add // chuẩn bị lưu lại trạng thái tại thời điểm hiện tại của dự án
- git reset // đổi ý chưa muốn lưu lại ở trạng thái hiện tại
- git commit -m // chính thức lưu cần lưu lại feature gì
- git log
- git log --oneline // check the commit history
- git checkout {branch name} // Quay lại thời điểm 
- git branch
- git checkout -b {branch name} // to create a new branch
- git merge {branch name} // để tổng hợp với branch master
- git branch -d {branch name} // to delete the branch
- git push // đẩy repository từ local lên remote
- git push -u {branchname} // push branch into github
- git remote add origin {repo url} // alias
- git push origin {branch name} đẩy branch lên github
- git clone {repo url}
- git fetch origin // chuẩn bị kéo cái branch trên github về.
- git checkout -b {branchname} origin/{branchname}
